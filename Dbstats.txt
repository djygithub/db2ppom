*
*     PROGRAM NAME - DBSTATS
*
*     FUNCTION     - RETRIEVE DB2 DATA BASE SERVICES ADDRESS SPACE
*                    STATISTICS VIA THE INSTRUMENTATION FACILITY
*                    INTERFACE
*
*     INPUT        - UPON ENTRY R1 POINTS AT THE FOLLOWING PARMS
*                    P1 - $PARM1   DSECT
*                         EYEBALL  DC    CL8'WORKAREA'
*                         WORKAREA DS    512F
*
*     REGISTERS    - R1 --> CALLER PARMLIST
*                    R3 --> WORKAREA SUPPLIED BY CALLER
*                    R4 --> IFCA
*                    R5 =   RETURN CODE FROM IFCA
*                    R6     WORK
*                    R7        REGISTERS
*                    R8              FOR
*                    R9                 MOVING DATA
*                    R11 --> RETURNED DB STATS
*                    R13    BASE REGISTER AND SAVE AREA POINTER
*
*     OUTPUT       - WORKAREA WILL BE FILLED IN WITH DSNDBM1 STATS
*
*     MACROS       - NONE
*
*     RETURN CODES - 0 - STATS WERE SUCCESSFULLY RETRIEVED
*                    4 - STATS WEREN'T SUCCESSFULLY RETRIEVED
*
*     (1)  GET INPUT PARAMETERS
*          SET UP IFCA
*
*     (2)  CALL DSNWLI FOR DB STATS
*
*     (3)  IF  CALL IS UNSUCCESSFUL
*
*     (4)      BRANCH TO NOGOOD AND SET RC=4
*
*          ELSE
*
*     (5)      ADJUST AND MOVE TIMESTAMP
*
*     (6)      MOVE SQL COUNTERS
*
*     (7)      MOVE SERVICE CONTROLLER DATA
*
*     (8)      MOVE BUFFER POOL INFO
*
*     (9)      MOVE IRLM LOCKING STATS
*
*     (10)     MOVE EDM POOL STATS
*
*     (11)     SET RC=0
*
*          ENDIF
*
*     (12) EXIT
*
**********************************************************************
DBSTATS   CSECT
          STM   R14,R12,12(R13)
          BAL   R14,80(R15)
          DC    18F'-1'
          ST    R13,4(R14)
          ST    R14,8(R13)
          LR    R13,R14
          USING DBSTATS+8,R13
*(1)
          L     R3,0(R1)             GET ADDRESS OF WORKAREA
          USING $PARM1,R3            SET UP ADDRESSABILITY
          XC    QXST#1(216),QXST#1
          XC    QTST#1(164),QTST#1
          XC    QBST1#1(240),QBST1#1
          XC    QBST1#60(36),QBST1#60
          XC    QBST2#1(240),QBST2#1
          XC    QBST2#60(36),QBST2#60
          XC    QBST3#1(240),QBST3#1
          XC    QBST3#60(36),QBST3#60
          XC    QBST4#1(240),QBST4#1
          XC    QBST4#60(36),QBST4#60
          XC    QTXA#1(96),QTXA#1
          XC    QISE#1(56),QISE#1
          XC    QIST#1(28),QIST#1
          XC    QBGL1#1(88),QBGL1#1
          XC    QBGL2#1(88),QBGL2#1
          XC    QBGL3#1(88),QBGL3#1
          XC    QBGL4#1(88),QBGL4#1
          XC    QTGS#1(76),QTGS#1
          L     R4,=F'-1'
          ST    R4,QTGS#1
          ST    R4,QBST1#1
          ST    R4,QBST2#1
          ST    R4,QBST3#1
          ST    R4,QBST4#1
          ST    R4,QBGL1#1
          ST    R4,QBGL2#1
          ST    R4,QBGL3#1
          ST    R4,QBGL4#1
          L     R4,=A(IFCAWORK)      GET ADDRESS OF IFCA WORKAREA
          USING IFCA,R4              SET UP ADDRESSABILITY
*(2)
          OI    PARMEND,X'80'        SET END OF PARMLIST
          L     R1,=A(PARMLIST)      POINT R1 AT PARMLIST
          L     R15,=V(DSNWLI)       POINT R15 AT DB2 INTERFACE
          BALR  R14,R15              GO GET STATS
*(3)
          L     R5,IFCARC1           GET RETURN CODE
          LTR   R5,R5                DID IT WORK?
*(4)
          BNZ   NOGOOD               BRANCH AROUND IF IT DIDN'T
*(5)
          LA    R11,RETNSPOT         R11--> START OF RECORD
          LA    R6,RETNSPOT+4        R6--> QWS1
          USING QWS1,R6              ESTABLISH ADDRESSABILITY
          L     R7,QWS10PSO          GET OFFSET TO PRODUCT SECTION
          LA    R8,0(R11,R7)         R8--> PRODUCT SECTION
          USING QWHS,R8              ESTABLISH ADDRESSABILITY
          MVC DBSSID(4),QWHSSSID     MOVE DB2 SUBSYSTEM ID
          LM    R0,R1,QWHSSTCK       GET TIMESTAMP
          SRDL  R0,4                 DIVIDE BY 16 TO AVOID 0C9
          STM   R0,R1,TIMESTMP       MOVE TO CALLER'S AREA
*(6)
          L     R7,QWS10R1O          GET OFFSET TO QXST
          LTR   R7,R7                ANY SQL ?
          BZ    QTSTSPOT             BRANCH AROUND IF NOT
          LA    R8,0(R11,R7)         R8--> QXST
          USING DSNDQXST,R8          ESTABLISH ADDRESSABILITY
          MVC   QXST#1(216),QXSELECT MOVE COUNTERS
*(7)
QTSTSPOT  L     R7,QWS10R2O          GET OFFSET TO QTST
          LA    R8,0(R11,R7)         R8--> QTST
          USING DSNDQTST,R8          ESTABLISH ADDRESSABILITY
          MVC   QTST#1(180),QTALLOCA MOVE COUNTERS
*(8)
          L     R7,QWS10R3O          GET OFFSET TO QBST
          LH    R9,QWS10R3L          GET LENGTH OF EACH ENTRY
          LH    R10,QWS10R3N         GET NUMBER OF ENTRIES
          LA    R8,0(R11,R7)         R8--> QBST BUFFER POOL 0
          USING DSNDQBST,R8          ESTABLISH ADDRESSABILITY
          MVC   QBST1#1(112),QBSTPID MOVE DATA
          MVC   QBST1#29(160),QBSTWFM MOVE DATA
          BCTR  R10,R0               SUBTRACT ONE
          LTR   R10,R10              ARE WE DONE?
          BZ    QTXASPOT             MAKE LIKE A BANNANA AND SPLIT
          LA    R8,0(R8,R9)          BUMP TO NEXT ENTRY
          MVC   QBST2#1(112),QBSTPID MOVE DATA
          MVC   QBST2#29(160),QBSTWFM MOVE DATA
          BCTR  R10,R0               SUBTRACT ONE
          LTR   R10,R10              ARE WE DONE?
          BZ    QTXASPOT             MAKE LIKE A BANNANA AND SPLIT
          LA    R8,0(R8,R9)          BUMP TO NEXT ENTRY
          MVC   QBST3#1(112),QBSTPID MOVE DATA
          MVC   QBST3#29(160),QBSTWFM MOVE DATA
          BCTR  R10,R0               SUBTRACT ONE
          LTR   R10,R10              ARE WE DONE?
          BZ    QTXASPOT             MAKE LIKE A BANNANA AND SPLIT
          LA    R8,0(R8,R9)          BUMP TO NEXT ENTRY
          MVC   QBST4#1(112),QBSTPID MOVE DATA
          MVC   QBST4#29(160),QBSTWFM MOVE DATA
*(9)
QTXASPOT  L     R7,QWS10R5O          GET OFFSET TO QTXA
          LA    R8,0(R11,R7)         R8--> QTXA
          USING DSNDQTXA,R8          ESTABLISH ADDRESSABILITY
          MVC   QTXA#1(88),QTXADEA   MOVE DATA
*(9A)
          L     R7,QWS10R4O          QIST
          LA    R8,0(R11,R7)         R8--> QIST
          USING DSNDQIST,R8
          MVC   QIST#1(28),QISTRLLM
*(10)
          L     R7,QWS10R6O          GET OFFSET TO QISE
          LA    R8,0(R11,R7)         R8--> QISE
          USING QISE,R8              ESTABLISH ADDRESSABILITY
          MVC   QISE#1(56),QISEFAIL  MOVE DATA
QBGLSPOT  L     R7,QWS10R7O          GET OFFSET
          LTR   R7,R7                ANYTHING TO LOOK AT?
          BZ    QTGSSPOT             GET OUT IF NOT
          LH    R9,QWS10R7L          GET LENGTH
          LH    R10,QWS10R7N         GET NUMBER
          LA    R8,0(R11,R7)         R8--> QBGL
          USING QBGL,R8
          MVC   QBGL1#1(92),QBGLGN   MOVE IT
          BCTR  R10,R0               DECREMENT
          LTR   R10,R10              ANY MORE?
          BZ QTGSSPOT                IF SO EXIT STAGE LEFT
          LA    R8,0(R8,R9)          BUMP TO NEXT ENTRY
          MVC   QBGL2#1(92),QBGLGN   MOVE IT
          BCTR  R10,R0               DECREMENT
          LTR   R10,R10              ANY MORE?
          BZ QTGSSPOT                IF SO EXIT STAGE LEFT
          LA    R8,0(R8,R9)          BUMP TO NEXT ENTRY
          MVC   QBGL3#1(92),QBGLGN   MOVE IT
          BCTR  R10,R0               DECREMENT
          LTR   R10,R10              ANY MORE?
          BZ QTGSSPOT                IF SO EXIT STAGE LEFT
          LA    R8,0(R8,R9)          BUMP TO NEXT ENTRY
          MVC   QBGL4#1(92),QBGLGN   MOVE IT
QTGSSPOT  L     R7,QWS10R8O          GET OFFSET
          LTR   R7,R7                ANYTHING?
          BZ    EXIT00               LEAVE IF NOT
          LA    R8,0(R11,R7)         R8--> QTGS
          USING QTGS,R8
          MVC   QTGS#1(76),QTGSLPLK
